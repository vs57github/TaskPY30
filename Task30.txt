a1 = int(input("Введите первый элемент прогрессии: "))
d = int(input("Введите разность прогрессии: "))
n = int(input("Введите количество элементов прогрессии: "))

# Создайте пустой массив для хранения элементов прогрессии
progression = []

# Заполните массив элементами прогрессии
for i in range(n):
    an = a1 + (i * d)
    progression.append(an)

# Выведите массив элементов прогрессии на экран
print("Элементы арифметической прогрессии:")
for element in progression:
    print(element)
